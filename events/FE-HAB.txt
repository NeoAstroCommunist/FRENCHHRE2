# Magarete of Tyrols heir dies
country_event = {
	id = flavor_hab.1010
	title = flavor_hab.1010.title
	desc = flavor_hab.1010.desc
	picture = DEATH_OF_HEIR_eventPicture
	
	fire_only_once = yes
	
	trigger = {
	    always = no
		tag = TIR
		is_female = yes
		has_ruler = "Margarete von Tirol - Görz"
		#has_heir = yes
	}
	
	mean_time_to_happen = {
		months = 20
		modifier = {
			factor = 0.75
			HAB = {
				dynasty = "von Habsburg"
				marriage_with = ROOT
			}
		}
		modifier = {
			factor = 0.75
			HAB = {
				dynasty = "von Habsburg"
				alliance_with = ROOT
			}
		}
		modifier = {
			factor = 0.5
			has_heir = no
		}
	}
	
	immediate = {
		if = {
			limit = {
				has_heir = yes
			}
			kill_heir = {}
			custom_tooltip = flavor_hab.ct.1
		}
		if = {
			limit = {
				has_heir = no
			}
			custom_tooltip = flavor_hab.ct.2
		}
	}
	
	option = {
		#Give the Land to the Habsburgs
		name = flavor_hab.1010.a
		ai_chance = { factor = 100 }
		set_global_flag = habsburgs_in_tyrol
		log = "FE-HAB:[GetYear]:[Root.GetName]:Magarete of Tyrols heir died, the next ruler will be Habsburg:flavor_hab.1010"
		define_heir = {
			#The next ruler is going to be a Habsburg
			claim = 75
			dynasty = "von Habsburg"
			age = 20
		}
		if = {
			#Tyrol is now part of the Habsburg domains
			limit = { HAB = { dynasty = "von Habsburg" } }
			create_alliance = HAB
			create_marriage = HAB
			log = "FE-HAB:[GetYear]:[Root.GetName]:alliance and marriage with Austria set:flavor_hab.1010"
		}
		if = {
			#The Witteslbachers have claims if the ruler is Margarete
			limit = { has_ruler = "Margarete von Tirol - Görz" }
			if = {
				limit = { BAV = { dynasty = "von Wittelsbach" } }
				BAV	= {
					add_casus_belli = {
						target = TIR
						type = cb_restore_personal_union
						months = 24
					}
				}
			}
			if = {
				limit = { BAW = { dynasty = "von Wittelsbach" } }
				BAW	= {
					add_casus_belli = {
						target = TIR
						type = cb_restore_personal_union
						months = 24
					}
				}
			}
			if = {
				limit = { BAX = { dynasty = "von Wittelsbach" } }
				BAX	= {
					add_casus_belli = {
						target = TIR
						type = cb_restore_personal_union
						months = 24
					}
				}
			}
			if = {
				limit = { BAY = { dynasty = "von Wittelsbach" } }
				BAY	= {
					add_casus_belli = {
						target = TIR
						type = cb_restore_personal_union
						months = 24
					}
				}
			}
			if = {
				limit = { BAZ = { dynasty = "von Wittelsbach" } }
				BAZ	= {
					add_casus_belli = {
						target = TIR
						type = cb_restore_personal_union
						months = 24
					}
				}
			}
		}
	}
	
	option = {
		#Retain our Dynasty
		name = flavor_hab.1010.b
		ai_chance = { factor = 0 }
		log = "FE-HAB:[GetYear]:[Root.GetName]:decided to reject the Habsburgs and keep their dynasty:flavor_hab.1010"
		if = {
			limit = { HAB = { dynasty = "von Habsburg" } }
			HAB = {
				add_casus_belli = {
					target = TIR
					type = cb_restore_personal_union
					months = 24
				}
			}
		}
	}
}

# Loss of the Habsburg dynasty
country_event = {
	id = flavor_hab.1011
	title = flavor_hab.1011.title
	desc = flavor_hab.1011.desc
	picture = KING_SICK_IN_BED_eventPicture
	
	major = yes
	is_triggered_only = yes
	
	trigger = {
	    always = no
		OR = {
			#You need to be part of the core Austrian Lands OR Tyrol that has Habsburg rulers through the Event
			tag = HAB
			tag = STY
			AND = {
				tag = TIR
				has_global_flag = habsburgs_in_tyrol
			}
		}
		is_at_war = no
		OR = {
			#One of the other core parts of Austria must be ruled by a Habsburg
			AND = {
				HAB = {
					dynasty = "von Habsburg"
					exists = yes
				}
				NOT = {
					tag = HAB
				}
			}
			AND = {
				STY = {
					dynasty = "von Habsburg"
					exists = yes
				}
				NOT = {
					tag = STY
				}
			}
			AND = {
				TIR = {
					dynasty = "von Habsburg"
					exists = yes
				}
				NOT = {
					tag = TIR
				}
			}
		}
		NOT = {
			dynasty = "von Habsburg"
			has_country_flag = no_habsburgs
			has_country_flag = habsburg_inheritance
		}
	}
	
	option = {
		#Give Austria back its lands
		name = flavor_hab.1011.a
		ai_chance = { factor = 1 }
		#Inheritance by HAB is prioritised over STY and TIR, if HAB doesn't exist but STY does, then STY will inherit, if HAB and STY don't exist, TIR will inherit
		log = "FE-HAB:[GetYear]:[Root.GetName]:decided to give back the lands to Habsburgs:flavor_hab.1011"

		if = {
			limit = {
				exists = HAB
				NOT = {	tag = HAB }
				HAB = {
					dynasty = "von Habsburg"
				}
			}
			HAB = {
				country_event = { id = flavor_hab.1012 }
			}
			log = "FE-HAB:[GetYear]:[Root.GetName]:inheriting country is Austria (HAB):flavor_hab.1011"
		}
		if = {
			limit = {
				exists = STY
				NOT = {
					exists = HAB
					tag = STY
				}
				STY = {
					dynasty = "von Habsburg"
				}
			}
			set_country_flag = habsburg_going_to_be_inherited
			STY = {
				country_event = { id = flavor_hab.1012 }
			}
			log = "FE-HAB:[GetYear]:[Root.GetName]:inheriting country is Styria (STY):flavor_hab.1011"
		}
		if = {
			limit = {
				exists = TIR
				NOT = {
					exists = HAB
					exists = STY
					tag = TIR
				}
				TIR = {
					dynasty = "von Habsburg"
				}
			}
			TIR = {
				country_event = { id = flavor_hab.1012 }
			}
		}
		set_country_flag = habsburg_inheritance
		log = "FE-HAB:[GetYear]:[Root.GetName]:inheriting country is Tirol (TIR):flavor_hab.1011"
	}
	
	option = {
		#Never!
		name = flavor_hab.1011.b
		ai_chance = { factor = 0 }
		set_country_flag = no_habsburgs
		log = "FE-HAB:[GetYear]:[Root.GetName]:decided not to give back the lands to Habsburgs:flavor_hab.1011"

		if = {
			limit = {
				tag = TIR
			}
			clr_global_flag = habsburgs_in_tyrol
		}
		if = {
			limit = {
				exists = HAB
				NOT = {	tag = HAB }
				HAB = {
					dynasty = "von Habsburg"
				}
			}
			HAB = {
				country_event = { id = flavor_hab.1013 }
			}
		}
		if = {
			limit = {
				exists = STY
				NOT = {	tag = STY }
				STY = {
					dynasty = "von Habsburg"
				}
			}
			STY = {
				country_event = { id = flavor_hab.1013 }
			}
		}
		if = {
			limit = {
				exists = TIR
				NOT = {	tag = TIR }
				TIR = {
					dynasty = "von Habsburg"
				}
			}
			TIR = {
				country_event = { id = flavor_hab.1013 }
			}
		}
		set_country_flag = habsburg_inheritance
	}
}

			}

		dynasty = "von Habsburg"
		is_at_war = no

			#Another of the core parts of Austria, OR a Tyrol that has gotten Habsburg per event, must have a Habsburg ruler

					exists = yes
				}
				NOT = {	tag = HAB }
			}
			AND = {
				STY = {
					dynasty = "von Habsburg"
					exists = yes
				}
				NOT = {	tag = STY }
			}
			AND = {
				TIR = {

				}
				NOT = {	tag = TIR }
			}
		}
		NOT = {
			has_country_flag = habsburg_inheritance
			}
	}
	
	immediate = {
		log = "FE-HAB:[GetYear]:[Root.GetName]:Habsburg ruler has died, other branch might acquire the land:flavor_hab.1014"

						}
					}
					set_country_flag = habsburg_going_to_be_inherited
					STY = {
						country_event = { id = flavor_hab.1015 }
					}
				}
				if = {
					limit = {
						exists = TIR
						NOT = {
							exists = HAB
							exists = STY
							tag = TIR
						}
						TIR = {
							dynasty = "von Habsburg"
						}
					}
					TIR = {
						country_event = { id = flavor_hab.1015 }
					}
				}
				set_country_flag = habsburg_inheritance
			}
			75 = {
				log = "FE-HAB:[GetYear]:[Root.GetName]:No aquisition of land by other Habsburg branch:flavor_hab.1014"
			}
		}
	}
